Reference

Ukazka predani hodnoty do funkce pomoci adresy

Je mozne hodnoty ve funkci obcerstvovat a pokud jich je vice pak se nevraci pres return, ale referenci

#include<iostream>
    using namespace std;


    // hodnoty jsou obcerstveny a navraceny referenci, tj. nejsou vraceny pres return
    // s vyhodou pouzit, pokud vracim vice hodnot (pres return jde jen jedna)
    
    void obcerstveniHodnot(int& hodnotaA, int& hodnotaB)
    {
    
        //obcerstvim hodnoty, tj. vynasobim je 10
        hodnotaA = hodnotaA * 10;
        hodnotaB = hodnotaB * 10;
    
    }
    
    
    // v nekterych pripadech je lepsi predavat promennou pomoci adresy
    void funkce(string& hodnota)
    {
        cout << "jsem ve funkci a predal jsem promennou pomoci adresy : " << endl;
        cout << hodnota << endl << endl;
        cout << "pricemz adresa je stejna jako v main : " << endl;
        cout << &hodnota << endl << endl;
    }
    
    
    // ZACATEK PROGRAMU ZDE
    int main()
    {
    
        string hodnotaMain = "toto je hodnota";
    
        cout << "jsem v main a adresa hodnoty je nasledujici : " << endl;
        cout << &hodnotaMain << endl << endl;
    
        // predam hodnotu do funkce
        funkce(hodnotaMain);
    
        //---------- obcerstveni hodnot pomoci reference --------------
        int hodnotaA = 10;
        int hodnotaB = 20;
    
        //zavolam funkci, data jiz nevracim
        obcerstveniHodnot(hodnotaA, hodnotaB);
    
        cout << "data jsou obcerstena, jsem ve funkci main : " << endl;
        cout << hodnotaA << endl;
        cout << hodnotaB << endl;
    
        return 0;
    }
