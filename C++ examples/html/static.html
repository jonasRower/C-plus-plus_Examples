Statická proměnná

Demonstruji na poctu volane tridy a zapisu dat do pole

#include <iostream>

    using namespace std;
    static int pole[4][4];
    
    void display() {
    
        static int counter = 0;
        cout << "funkce byla zavolana " << ++counter << " krat" << endl;;
    
    }
    
    
    //funkce zapisuje data do statickeho 2D pole
    void zapisDoPole(int radek, int sloupec, int hodnota) {
    
        pole[radek][sloupec] = hodnota;
    
    }
    
    
    //funkce tiskne data z 2D pole
    void tiskniPole() {
    
        for (int i = 0; i < 4; ++i) {
            for (int j = 0; j < 4; ++j) {
                printf("%3d ", pole[i][j]);
            }
            printf("\n");
        }
    
    }
    
    
    //funkce main
    int main()
    {
        //pocita pocet funkci, ktere zavolal
        display();
        display();
        display();
    
        cout << endl;
    
        //zapise nektere prvky do pole
        zapisDoPole(0, 0, 1);
        zapisDoPole(0, 2, 2);
        zapisDoPole(2, 0, 3);
    
        //vytiskne pole
        tiskniPole();
    
        cout << endl << "evidentne pole bylo naplneno a vytisknuto spravne" << endl;
    
        return 0;
    }
