Ukázka struktury


    #include <iostream>
    struct sablonaStruktury // šablona struktury
    {
        char retezec[20];
        float cislo;
    };
    
    
    int main()
    {
        using namespace std;
        sablonaStruktury struktura1 =
        {
            "Retezec struktury 1", 
            1.88 						
                    
        }; 								
    
        sablonaStruktury struktura2 =
        {
            "Retezec struktury 2",
            3.12,
        }; 	
    
        cout << "retezce v obou strukturach\n";
        cout << struktura1.retezec;
        cout << "\n";
        cout << struktura2.retezec;
        cout << "\n\n";
        cout << "cisla v obou strukturach\n";
        cout << struktura2.cislo;
        cout << "\n";
        cout << struktura1.cislo;
        cout << "\n";
    
        return 0;
    }
