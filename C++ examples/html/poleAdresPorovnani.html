Pole adres - porovnání

Výstup programu zobrazuje tabulku, kde se porovnávají velikosti sizeOf a rozdilu adres mezi prvky pole.

Z toho je možné usoudit, jaké prvky patří do pole a které již ne.

Příklad také ilustruje pointerovou aritmetiku - adresy jsou rozloženy po 4 bytech.

#include <iostream>
    #include <string>
    #include <iomanip>
    #include <sstream>
    
    using namespace std;
    
    //funkce jen konvertuje na decimalni vyraz
    int konvertujNaDec(int *adresaHex)
    {
        int adresaDec;
        std::stringstream stream;
        stream << adresaHex;
        stream >> std::hex >> adresaDec;
    
        return(adresaDec);
    }
    
    
    int main()
    {
        using namespace std;
    
        int poleInt[4];
        int* poleAdres[4];
        int adresaDec[4];
    
        poleInt[0] = 0;
        poleInt[1] = 1;
        poleInt[2] = 2;
        poleInt[3] = 3;
    
        //adresy hexadecimalni
        poleAdres[0] = &poleInt[0];
        poleAdres[1] = &poleInt[1];
        poleAdres[2] = &poleInt[2];
        poleAdres[3] = &poleInt[4];  //zde je umyslne &poleInt[4] - adresa je jiz mimo rozsah pole
    
        //adresy dekadicke
        adresaDec[0] = konvertujNaDec(poleAdres[0]);
        adresaDec[1] = konvertujNaDec(poleAdres[1]);
        adresaDec[2] = konvertujNaDec(poleAdres[2]);
        adresaDec[3] = konvertujNaDec(poleAdres[3]);
    
    
        //tiskne hlavicku tabulky
        cout << "  poleInt[i]  " << "  poleAdres[i]  " << "  adresaDec[i]  " << "  adresaDec([i] - [i+1])   " << "sizeOf(poleInt[i])  " << "   poznamka"<< endl;
        
        //tiskne data a dopocitava adresu decimalni
        cout << "      " << poleInt[0] << "           " << poleAdres[0] << "        " << adresaDec[0] << "                  " << " " << "                    " << sizeof(poleInt[1]) << "  " << "  " << endl;
        cout << "      " << poleInt[1] << "           " << poleAdres[1] << "        " << adresaDec[1] << "                  " << adresaDec[1] - adresaDec[0] << "                    " << sizeof(poleInt[1]) << "           " << "prvek je v poli" << endl;
        cout << "      " << poleInt[2] << "           " << poleAdres[2] << "        " << adresaDec[2] << "                  " << adresaDec[2] - adresaDec[1] << "                    " << sizeof(poleInt[2]) << "           " << "prvek je v poli" << endl;
        cout << "      " << poleInt[3] << "           " << poleAdres[3] << "        " << adresaDec[3] << "                  " << adresaDec[3] - adresaDec[2] << "                    " << sizeof(poleInt[3]) << "           " << "prvek neni v poli" << endl;
        cout << "  " << poleInt[4] << "      " << poleAdres[4] << "      " << adresaDec[4] << "            " << adresaDec[4] - adresaDec[3] << "                " << sizeof(poleInt[4]) << "           " << "index je mimo pole" << endl;
        
    
        return 0;
    }
