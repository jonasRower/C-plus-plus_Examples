Rozdíl mezi strlen a sizeof

sizeof - velikost obsazené paměti

strlen - velikost alokované paměti

#include <iostream>
    #include <string> 
    int main()
    {
    
        using namespace std;
        char poleZnaku7[7];
        poleZnaku7[0] = 'j';
        poleZnaku7[1] = 'a';
        poleZnaku7[2] = 'g';
        poleZnaku7[3] = 'u';
        poleZnaku7[4] = 'a';
        poleZnaku7[5] = 'r';
        poleZnaku7[6] = NULL;
    
    
        char poleZnaku20[20];
        poleZnaku20[0] = 'j';
        poleZnaku20[1] = 'a';
        poleZnaku20[2] = 'g';
        poleZnaku20[3] = 'u';
        poleZnaku20[4] = 'a';
        poleZnaku20[5] = 'r';
        poleZnaku20[6] = NULL;
    
    
        char poleZnaku6[6];
        poleZnaku6[0] = 'j';
        poleZnaku6[1] = 'a';
        poleZnaku6[2] = 'g';
        poleZnaku6[3] = 'u';
        poleZnaku6[4] = 'a';
        poleZnaku6[5] = 'r';
    
        cout << "char poleZnaku1[7]  <-- obsahuje presny pocet indexu v poli\n";
        cout << " " << poleZnaku7 << "\n""\n";
    
        cout << "char poleZnaku1[20]  <-- obsahuje nadmerny pocet indexu v poli\n";
        cout << " " << poleZnaku7 << "\n""\n";
        cout << "je-li na poslednim znaku NULL, pak mu to necini potize\n\n";
    
        cout << "------------------------------------------------\n";
        cout << "char poleZnaku6[6]  <-- neobsahuje NULL na konci\n";
        cout << "i kdyz odpovida poctu znaku, vypisuje hlouposti\n";
        cout << " " << poleZnaku6 << "\n""\n";
    
    
        cout << "------------------------------------------------\n";
        cout << "to take vysvetluje sizeof a strlen\n";
        cout << "sizeof popisuje velikost obsazeneho pole (obsazenou pamet)\n";
        cout << "zatimco strlen velikost alokovane pameti\n\n";
    
        cout << "je-li tudiz retezec ukoncen znakem NULL, pak je vse OK\n";
    
        cout << "\nsizeof(poleZnaku7) = " << sizeof(poleZnaku7);
        cout << "\nstrlen(poleZnaku7) = " << strlen(poleZnaku7);
    
        cout << "\nsizeof(poleZnaku20) = " << sizeof(poleZnaku20);
        cout << "\nstrlen(poleZnaku20) = " << strlen(poleZnaku20);
    
        cout << "\n\nneni-li retezec ukoncen znakem NULL, pak je naalokovana vyssi pamet nez je potrebna a pri vypisu se i zobrazuji nevalidni znaky\n";
    
        cout << "\nsizeof(poleZnaku6) = " << sizeof(poleZnaku6);
        cout << "\nstrlen(poleZnaku6) = " << strlen(poleZnaku6);
    
    
    
        return 0;
    }
