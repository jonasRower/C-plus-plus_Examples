#include <iostream>
    #include <string>

    using namespace std;
    
    class Variables
    {
    public:
    
        //musi byt deklarovany pointer, jinak nejdou nacitat data v main()
        int* Items;
    
        Variables()
        {
            Items = new int[10];
            Items[0] = 1;
            Items[1] = 2;
            Items[2] = 3;
            Items[3] = 4;
            Items[4] = 5;
            Items[5] = 6;
        }
    
    };
    
    int main()
    {
        Variables VO;
        cout << VO.Items[0] << endl;
    }
