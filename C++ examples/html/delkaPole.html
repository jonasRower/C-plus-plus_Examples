Ukázka délky pole

Délka pole se zjišťuje pomocí sizeof nebo přes pointer

#include <iostream>
    using namespace std;

    int main() {
        int arr[] = { 10,20,30,40,50,60 };
        int arrSize = sizeof(arr) / sizeof(arr[0]);
        int arrSizePointer = *(&arr + 1) - arr;
        cout << "Velikost pole je :   " << arrSize << endl;
        cout << "Velikost pole ziskana pres pointer :   " << arrSizePointer << endl;
        return 0;
    }
