Ukázka kódu pole

Pomocí sizeof se zjistí velikost pole

#include <iostream>
    int main()
    {
        using namespace std;
        int pole[3]; 	
        pole[0] = 7;
        pole[1] = 8;
        pole[2] = 6;
    
        int velikostPole = sizeof pole;
        int velikostJednohoPrvku = sizeof pole[0];
        int pocetPrvku = velikostPole / velikostJednohoPrvku;
    
        cout << "\nVelikost pole = " << velikostPole;
        cout << " bajtu.\n";
        cout << "Velikost jednoho prvku = " << velikostJednohoPrvku;
        cout << " bajty.\n";
        cout << "Pocet prvku v poli = " << pocetPrvku;
        cout << "\n";
    
        return 0;
    }
