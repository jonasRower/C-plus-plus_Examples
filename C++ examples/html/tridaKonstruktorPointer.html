Předání pointeru do konstruktoru

#include <iostream>
    #include <string>
    using namespace std;
    
    class  prvniTrida {
    public:
    
        string* pole = new string[3];
    
        //konstructor
        prvniTrida() {
            pole[0] = "Simon";
            pole[1] = "Peter";
            pole[2] = "Dave";
        }
    
    
        //geter
        string* getNames() {
            return pole;
        }
    
    };
    
    
    class  druhaTrida {
    public:
    
        //konstructor - vypise data
        druhaTrida(string* pole) {
            cout << "pole[1] " << pole[1];
        }
    
    };
    
    
    int main() {
    
        using namespace std;
    
        //inicializuje prvni tridu
        prvniTrida mojePrvniTrida;
    
        //ziska data z prvni tridy
        string* pole = new string[3];
        pole = mojePrvniTrida.getNames();
    
        //inicializuje druhou tridu
        druhaTrida mojeDuhaTrida(pole);
    
    
        return 0;
    }
