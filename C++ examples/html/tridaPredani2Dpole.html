Předání 2D pole

#include <iostream>
    #include <string>
    using namespace std;
    
    class  pole2D {
    public:
    
        // deklarace pole 2D
        int** array2D;
    
        pole2D(){
    
            array2D = new int* [3];
    
            array2D[0] = new int[2];
            array2D[0][0] = 5;
            array2D[0][1] = 13;
    
            array2D[1] = new int[2];
            array2D[1][0] = 5;
            array2D[1][1] = 15;
    
            array2D[2] = new int[2];
            array2D[2][0] = 8;
            array2D[2][1] = 18;
    
        }
    
        int** get2DArray() {
    
            return array2D;
    
        }
    
    };
    
    
    int main() {
    
        pole2D arr2D;
    
        int** my2DArray;
        my2DArray = arr2D.get2DArray();
    
        for (int h = 0; h < 3; h++)
        {
            for (int w = 0; w < 2; w++)
            {
                cout << my2DArray[h][w] << "  ";
            }
            cout << "\n";
        }
    
        return 0;
    
    }
