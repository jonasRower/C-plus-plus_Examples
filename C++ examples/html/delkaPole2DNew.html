Ukázka délky pole

Pole je deklarováno pomocí klíčového slova new

Zjišťuje se počet řádků i počet sloupců 2D pole

#include <iostream>
    #include <stdio.h>

    using namespace std;
    
    int main() {
    
        int** array2D;
    
        array2D = new int* [3];
    
        array2D[0] = new int[2];
        array2D[0][0] = 5;
        array2D[0][1] = 13;
    
        array2D[1] = new int[2];
        array2D[1][0] = 5;
        array2D[1][1] = 15;
    
        array2D[2] = new int[2];
        array2D[2][0] = 8;
        array2D[2][1] = 18;
    
    
        //zjisti velikost pole
        int pocetRadku = _msize(array2D) / sizeof(int);
        int pocetSloupcu = _msize(array2D[0]) / sizeof(int);
    
    
        //vypise velikost pole
        cout << "pocet radku    : " << pocetRadku << endl;
        cout << "pocet sloupcu  : " << pocetSloupcu << endl;
    
        return 0;
    }
