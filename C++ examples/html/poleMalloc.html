Pole - malloc

Vytvoření pole pomocí malloc

#include <stdlib.h> 
    #include <stdio.h> 

    int main(int argc, char const* argv[]) {
    
        int N = 3, count = 0;
        int** A;
    
        A = (int**)malloc(sizeof(int*) * N);
    
        for (int i = 0; i < N; ++i) {
            *(A + i) = (int*)malloc(sizeof(int) * N);
        }
    
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < N; ++j) {
                A[i][j] = count++;
            }
        }
    
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < N; ++j) {
                printf("%3d ", A[i][j]);
            }
            printf("\n");
        }
    
    
        return 0;
    }
