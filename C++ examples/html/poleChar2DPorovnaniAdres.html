Pole char 2D - porovnání adres

Výsledek programu zobrazuje tabulku, kde jsou porovnány adresy jednotlivých prvků pole char.

#include <iostream>
    #include <string>
    #include <iomanip>
    #include <sstream>
    #include <cstdint>
    
    using namespace std;
    
    //funkce jen konvertuje na decimalni vyraz
    int konvertujNaDec(char** adresaHex)
    {
        int adresaDec;
        std::stringstream stream;
        stream << adresaHex;
        stream >> std::hex >> adresaDec;
    
        return(adresaDec);
    }
    
    int main()
    {
        using namespace std;
    
        int poleInt[4];
        char* adresyCH[5];
        char** adresyCH2[5];
        int adresaDec[5];
    
        char *h;
    
    
        char* CH;
    
        CH = (char*)malloc(sizeof(char) * 5);
    
        CH[0] = 'c';
        CH[1] = 'a';
        CH[2] = 'r';
    
        //posledni znak musi byt NULL, jinak vypisuje nesmysly
        CH[3] = NULL;
        
        //adresy - ukazuje na prvni index od ktereho uvazuje radek
        adresyCH[0] = &CH[0];
        adresyCH[1] = &CH[1];
        adresyCH[2] = &CH[2];
        adresyCH[3] = &CH[3];
        adresyCH[4] = &CH[4];
    
        //adresy - ukazuje na skutecnou adresu daneho znaku
        adresyCH2[0] = &adresyCH[0];
        adresyCH2[1] = &adresyCH[1];
        adresyCH2[2] = &adresyCH[2];
        adresyCH2[3] = &adresyCH[3];
        adresyCH2[4] = &adresyCH[4];
    
        //adresy dekadicke
        adresaDec[0] = konvertujNaDec(adresyCH2[0]);
        adresaDec[1] = konvertujNaDec(adresyCH2[1]);
        adresaDec[2] = konvertujNaDec(adresyCH2[2]);
        adresaDec[3] = konvertujNaDec(adresyCH2[3]);
        adresaDec[4] = konvertujNaDec(adresyCH2[4]);
        
    
     
        cout << " CH[i] " << " adresyCH[i] " << "   adresyCH2[i] " << "   adresaDec[i] " << " adresaDec([i+1] - [i]) " << endl;
        cout << "  " << CH[0] << "         " << adresyCH[0] << "         " << adresyCH2[0] << "         " << adresaDec[0] << "              " << "                        " << endl;
        cout << "  " << CH[1] << "         " << adresyCH[1] << "          " << adresyCH2[1] << "         " << adresaDec[1] << "               " << adresaDec[1] - adresaDec[0] << endl;
        cout << "  " << CH[2] << "         " << adresyCH[2] << "           " << adresyCH2[2] << "         " << adresaDec[2] << "               " << adresaDec[2] - adresaDec[1] << endl;
        cout << "  " << CH[3] << "         " << adresyCH[3] << "             " << adresyCH2[3] << "         " << adresaDec[3] << "               " << adresaDec[3] - adresaDec[2] << endl;
        
        cout << endl << endl << " je otazkou jake znaky jsou na mezilehlych bytech, kdyz velikost char je : " << sizeof(char) << " byte" << endl;
        
    
        return 0;
    
    }
