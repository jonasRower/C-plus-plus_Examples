Statické pole stringu

Pole stringu je zde nahrazeno const char*

Statická data jsou uvnitř třídy

Jednotlivé položky pole je potřeba překonvertovat ze stringu pomocí const char* tempCH = zapisStr2.c_str();

Do dané metody je třeba předat adresu dat, jinak se zapisují data jiná

#include <iostream>
    #include <string>

    using namespace std;
    
    
    class tridaZapisDataDoPole {
    public:
    
        static int poradiVolani;
        static const char* pole[10];
    
        //konstruktor bezparametru
        tridaZapisDataDoPole()
        {
            //nedela nic
            //instanci tohoto konstruktoru je volana metoda "TiskPole"
        }
    
        //konstruktor pretizeny
        //je potreba predavat data pomoci adresy, jinak se do pole const char* vkladaji jina data
        tridaZapisDataDoPole(string& zapisStr2)
        {
            poradiVolani++;
    
            const char* tempCH = zapisStr2.c_str();
            pole[poradiVolani] = tempCH;
            
        }
    
        void TiskPole() {
    
            for (int i = 0; i < poradiVolani+1; i++)
            {
                 cout << pole[i] << endl;
            }
        }
    
    };
    
    
    class tridaTiskPole {
    public:
    
        // tisk pole je mozny jen tak, 
        // ze vytvorim novou instanci tridy, kde je pole uchovano
        // a vytisknu pole pomovi metody v teze tride
    
        //odsud nelze statická data tisknout
    
        tridaTiskPole() {
            tridaZapisDataDoPole instanceProTiskPole;
            instanceProTiskPole.TiskPole();
        }
    
    };
    
    
    // Inicializace statickeho clena tridy
    int tridaZapisDataDoPole::poradiVolani = -1;
    const char* tridaZapisDataDoPole::pole[10];
    
    
    int main() {
    
        string zapisStr1 = "zapis data 1";
        string zapisStr2 = "zapis data 2";
    
        tridaZapisDataDoPole zapisDataDoPole(zapisStr1);
        tridaZapisDataDoPole zapisDataDoPole2(zapisStr2);
        tridaTiskPole tiskPole;
    
        return 0;
    }
