Převod z string na char[] + rozdíl mezi strlen / sizeof

sizeof - velikost obsazené paměti

strlen - velikost alokované paměti

#include <iostream>
    #include <string>

    using namespace std;
    
    int main() {
    
        //prevadim string "cat" na pole znaku
        string temp = "cat";
       
        //pole char deklaruji s delkou 10
        char tab2[10];
    
        //zapisu data do pole char
        strncpy_s(tab2, temp.c_str(), sizeof(tab2));
        tab2[sizeof(tab2) - 1] = 0;
    
        //zde demonstruji rozdil mezi strlen a sizeof
        //strlen udava delku retezce v poli char, tedy jen zabrana pismena
        cout << "strlen :  " << strlen(tab2) << endl;
       
        //zatimco sizeof udava celkovou na-alokovanou delku, tedy 10
        cout << "sizeof :  " << sizeof(tab2) << endl;
    
        //nize tiskneme znaky s poctem cyklu do sizeof(tab2)
        for (int i = 0; i < sizeof(tab2); ++i) {
            printf("%3c ", tab2[i]);
            printf("\n");
        }
    }
