Ukázka délky pole

Zjišťuje se počet řádků i počet sloupců 2D pole

#include <iostream>
    using namespace std;

    int main() {
    
        int ary[][5] = { {1, 2, 3, 4, 5},
                       {6, 7, 8, 9, 0}
        };
    
        int rows = sizeof ary / sizeof ary[0];
        int cols = sizeof ary[0] / sizeof(int);
    
        int rowsPointer = *(&ary + 1) - ary;
    
        cout << "Pocet radku je   :   " << rows << endl;
        cout << "Pocet sloupcu je :   " << cols << endl;
    
        cout << endl;
        cout << "Pocet radku pres pointer je   :   " << rowsPointer << endl;
    
        return 0;
    
    }
