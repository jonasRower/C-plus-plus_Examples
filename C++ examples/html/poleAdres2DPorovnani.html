Pole integeru - porovnání adres jednotlivých prvků

Výstup programu zobrazuje tabulku, kde se porovnávají adresy jednotlivých prvků pole.

Pole je alokováno pomocí malloc.

Příklad také ilustruje pointerovou aritmetiku - adresy jsou rozloženy po 4 bytech.

#include <iostream>
    #include <string>
    #include <iomanip>
    #include <sstream>
    
    
    //funkce jen konvertuje na decimalni vyraz
    int konvertujNaDec(int* adresaHex)
    {
        int adresaDec;
        std::stringstream stream;
        stream << adresaHex;
        stream >> std::hex >> adresaDec;
    
        return(adresaDec);
    }
    
    
    //funkce jen konvertuje na decimalni vyraz (pretizena funkce)
    int konvertujNaDec(int** adresaHex)
    {
        int adresaDec;
        std::stringstream stream;
        stream << adresaHex;
        stream >> std::hex >> adresaDec;
    
        return(adresaDec);
    }
    
    
    
    int main() {
    
        using namespace std;
    
        int R = 3, count = 0;
        int S = 5;
        int** A;
    
        //adresy hexadecimalni
        int* adresyRadku[3];
    
        //adresy dekadicke
        int adresyRadkuDec[3];
        
        A = (int**)malloc(sizeof(int*) * R);
        
       
        for (int i = 0; i < R; ++i) {
    
            //kazda adresa *(A + i) se naalokuje polem integeru
            *(A + i) = (int*)malloc(sizeof(int) * S);
    
            //ulozi adresu radku do pole
            adresyRadku[i] = *(A + i);
    
            //prekonvertuje adresu na decimalni
            adresyRadkuDec[i] = konvertujNaDec(adresyRadku[i]);
    
        }
    
        //adresy hexadecimalni
        int** adresyA0[9];
        int* adresyA00[6];
        int* adresyA01[6];
    
        //adresy dekadicke
        int adresyA0Dec[9];
        int adresyA00Dec[6];
        int adresyA01Dec[6];
    
        for (int i = 0; i < 9; ++i) {
            adresyA0[i] = &A[i];
            adresyA0Dec[i] = konvertujNaDec(adresyA0[i]);
        
        }
    
        for (int i = 0; i < 6; ++i) {
            adresyA00[i] = &A[0][i];
            adresyA00Dec[i] = konvertujNaDec(adresyA00[i]);
     
        }
    
        for (int i = 0; i < 6; ++i) {
            adresyA01[i] = &A[1][i];
            adresyA01Dec[i] = konvertujNaDec(adresyA01[i]);
    
        }
     
    
    
        cout << endl << "Pole integeru muze vypadat nasledovne :" << endl;
        for (int i = 0; i < R; ++i) {
            for (int j = 0; j < S; ++j) {
                A[i][j] = count++;
                printf("%3d ", A[i][j]);
            }
            printf("\n");
        }
    
   
        //tiskne hlavicku tabulky
        cout << endl << "------------------------------------------------------ Tabulka A  -------------------------------------------------------" << endl;
        cout << endl << "    radek " << "                  adresyRadku[i]  " << "  adresyRadkuDec[i]  " <<  "  adresaDec([i+1] - [i])   " << "sizeof(int) * 5  " << endl;
        
        //tiskne data
        cout << "      1                        " << adresyRadku[0] << "           " << adresyRadkuDec[0] << "                 " << " " << "                    " << sizeof(int) * S << "  " << "  " << endl;
        cout << "      2                        " << adresyRadku[1] << "           " << adresyRadkuDec[1] << "                 " << adresyRadkuDec[1] - adresyRadkuDec[0] << "                " << sizeof(int) * S << endl;
        cout << "      3                        " << adresyRadku[2] << "           " << adresyRadkuDec[2] << "                 " << adresyRadkuDec[2] - adresyRadkuDec[1] << "                " << sizeof(int) * S << endl;

        
        cout << endl << " je evidentni, ze zadne hodnoty se nedaji pouzit. Adresy jsou nahodne. Jedine co lze pouzit je adresa na 1. radku." << endl;

        //tiskne hlavicku tabulky
        cout << endl << endl << "------------------------------------------------------ Tabulka B  -------------------------------------------------------   " << endl;
        cout << endl << "     pole   " << "hodnota v poli" << "  adresyRadku[i]  " << "  adresyRadkuDec[i]  " << "  adresaDec([i+1] - [i])  " << "sizeof(int)  " << "  poznamka  " << endl;

        
        //tiskne data
        cout << "  " << "   A[0]" << "      " << A[0] << "        " << adresyA0[0] << "           " << adresyA0Dec[0] << "                 " << " " << "                    " << sizeof(int) << "      " << "  hodnota v poli = adresa 1. radku - tab A " << endl;
        cout << "  " << "   A[1]" << "      " << A[1] << "        " << adresyA0[1] << "           " << adresyA0Dec[1] << "                 " << adresyA0Dec[1] - adresyA0Dec[0] << "                    " << sizeof(int) << "        hodnota v poli = adresa 2. radku - tab A " << endl;
        cout << "  " << "   A[2]" << "      " << A[2] << "        " << adresyA0[2] << "           " << adresyA0Dec[2] << "                 " << adresyA0Dec[2] - adresyA0Dec[1] << "                    " << sizeof(int) << "        hodnota v poli = adresa 3. radku - tab A " << endl;
        cout << "  " << "   A[3]" << "      " << A[3] << "        " << adresyA0[3] << "           " << adresyA0Dec[3] << "                 " << adresyA0Dec[3] - adresyA0Dec[2] << "                    " << sizeof(int) << "        hodnota v poli mimo rozsah pole " << endl;
        cout << "  " << "   A[4]" << "      " << A[4] << "        " << adresyA0[4] << "           " << adresyA0Dec[4] << "                 " << adresyA0Dec[4] - adresyA0Dec[3] << "                    " << sizeof(int) << "        hodnota v poli mimo rozsah pole " << endl;
        cout << "  " << "   A[5]" << "      " << A[5] << "        " << adresyA0[5] << "           " << adresyA0Dec[5] << "                 " << adresyA0Dec[5] - adresyA0Dec[4] << "                    " << sizeof(int) << "        hodnota v poli mimo rozsah pole " << endl;
        cout << "  " << "   A[6]" << "      " << A[6] << "        " << adresyA0[6] << "           " << adresyA0Dec[6] << "                 " << adresyA0Dec[6] - adresyA0Dec[5] << "                    " << sizeof(int) << "        hodnota v poli mimo rozsah pole " << endl;
        cout << "  " << "   A[7]" << "      " << A[7] << "        " << adresyA0[7] << "           " << adresyA0Dec[7] << "                 " << adresyA0Dec[7] - adresyA0Dec[6] << "                    " << sizeof(int) << "        hodnota v poli mimo rozsah pole " << endl;
        cout << "  " << "   A[8]" << "      " << A[8] << "        " << adresyA0[8] << "           " << adresyA0Dec[8] << "                 " << adresyA0Dec[8] - adresyA0Dec[7] << "                    " << sizeof(int) << "        hodnota v poli mimo rozsah pole " << endl;

        cout << "  " << "A[0][0]" << "         " << A[0][0] << "            " << adresyA00[0] << "           " << adresyA00Dec[0] << "              " << adresyA00Dec[0] - adresyA0Dec[8] << "                  " << sizeof(int) << "        adresyRadku = hodnota v poli A[0] " << endl;
        cout << "  " << "A[0][1]" << "         " << A[0][1] << "            " << adresyA00[1] << "           " << adresyA00Dec[1] << "                 " << adresyA00Dec[1] - adresyA00Dec[0] << "                    " << sizeof(int) << "        adresyRadku je posunuta o 4 byte - OK" << endl;
        cout << "  " << "A[0][2]" << "         " << A[0][2] << "            " << adresyA00[2] << "           " << adresyA00Dec[2] << "                 " << adresyA00Dec[2] - adresyA00Dec[1] << "                    " << sizeof(int) << "        adresyRadku je posunuta o 4 byte - OK" << endl;
        cout << "  " << "A[0][3]" << "         " << A[0][3] << "            " << adresyA00[3] << "           " << adresyA00Dec[3] << "                 " << adresyA00Dec[3] - adresyA00Dec[2] << "                    " << sizeof(int) << "        adresyRadku je posunuta o 4 byte - OK" << endl;
        cout << "  " << "A[0][4]" << "         " << A[0][4] << "            " << adresyA00[4] << "           " << adresyA00Dec[4] << "                 " << adresyA00Dec[4] - adresyA00Dec[3] << "                    " << sizeof(int) << "        adresyRadku je posunuta o 4 byte - OK" << endl;
        
        cout << "  " << "A[1][0]" << "         " << A[1][0] << "            " << adresyA01[0] << "           " << adresyA01Dec[0] << "               " << adresyA01Dec[0] - adresyA00Dec[4] << "                  " << sizeof(int) << "        adresyRadku = hodnota v poli A[1] " << endl;
