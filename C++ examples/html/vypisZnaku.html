Výpis z char i string

Jak na pole char tak i na String se jednotlive znaky dají načítat pomocí indexů

#include <iostream>
    #include <string> 
    int main()
    {
        using namespace std;						
        char charr2[20] = "jaguar"; 										
        string str2 = "panter"; 			
    
        cout << "Zde jsou kockovite selmy:\n";
        cout << " " << charr2 << "\n"
            << " " << str2 		
            << endl;
    
        cout << "\n---------------------------------------------------------\n";
        cout << "Lze vypsat pouze dane pismeno jak z char, tak i z string:\n\n";
    
        cout << "Treti pismeno v " << charr2 << " je "
            << charr2[2] << endl;
        cout << "Treti pismeno v " << str2 << " je "
            << str2[2] << endl; 	
    
    
        return 0;
    }
