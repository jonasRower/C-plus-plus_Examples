Statická proměnná

Statická proměnná odkazuje na index řádku pole na který bude zapisováno

#include <iostream>
    #include <cstdlib>

    using namespace std;
    
    
    class trida {
    public:
        static int poradiVolani;
        float* pole;
    
        // Konstruktor
        trida(int pocetPrvku) {
            alokujPole(pocetPrvku);
        }
    
        void alokujPole(int pocetPrvku)
        {
            pole = (float*)malloc(pocetPrvku * sizeof(float));
            for (int i = 0; i < pocetPrvku; i++)
            {
                pole[i] = i;
            }
        }
    
        void zapisIndexDoPole() 
        {
            poradiVolani++;
            pole[poradiVolani] = poradiVolani*10;
        }
    
        void vytiskniPole(int pocetPrvku)
        {
            for (int i = 0; i < pocetPrvku; i++)
            {
                cout << pole[i] << endl;
            }
        }
    
    };
    
    
    // Inicializace statického člena třídy
    int trida::poradiVolani = -1;
    
    int main(void) {
    
        int pocetPrvku = 5;
    
        //konstruktor vola alokaci pole
        trida trida(pocetPrvku);
        
        //kazdym volanim zapise index volani do pole
        trida.zapisIndexDoPole();
        trida.zapisIndexDoPole();
    
        //vytiskne pole
        trida.vytiskniPole(pocetPrvku);
    
    
        return 0;
    }
