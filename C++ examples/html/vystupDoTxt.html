Výstup do txt

Do textového souboru vypíše výstup

#include <iostream>
    #include <fstream>
    using namespace std;
    
    
    int main()
    {
    
        string vystup = "";
    
        string cars[4] = { "Volvo", "BMW", "Ford", "Mazda" };
        for (int i = 0; i < 4; i++) {
            vystup = vystup + cars[i] + "\n";
        }
    
        ofstream myfile;
        myfile.open("C:\\Users\\jonas\\source\\repos\\Pokusy\\example.txt");
        myfile << vystup;
        myfile.close();
        return 0;
    
    }
