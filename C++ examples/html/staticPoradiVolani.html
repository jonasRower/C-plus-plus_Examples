Statická proměnná

Využití pro zjištění kolikrát byla daná metoda volána.

I když se může jednat o jinou instanci třídy

#include <iostream>

    using namespace std;

    class trida {
    public:
        static int poradiVolani;
    
        // Konstruktor
        trida() {
            //Zde mohou byt nejaka vychozi data
        }
    
        void vratPoradiVolani() {
    
            poradiVolani++;
            cout << "Poradi volani je : " << poradiVolani << endl;
    
        }
        
    };
    
    // Inicializace statického člena třídy
    int trida::poradiVolani = 0;
    
    int main(void) {
    
        // i kdyz jsou 2 tridy poradiVolani se pocita jen jednou
        trida tridaA;
        trida tridaB;
    
        tridaA.vratPoradiVolani();
        tridaB.vratPoradiVolani();
        tridaA.vratPoradiVolani();
        tridaB.vratPoradiVolani();
    
    
        return 0;
    }
