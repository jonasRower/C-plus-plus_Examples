Statické pole

Porovnává pole int String a const char

První část zobrazuje data generovaná konstruktorem, druhá část změnu položky, a třetí změnu, přičemž položka je konvertována z float

U statických dat je potřeba dát pozor na to, aby se data definovala pouze 1x - aby se s voláník konstruktoru nepřepisovala zpět na originál

#include <iostream>
    #include <cstdlib>
    #include <sstream> //konverze float na string
    
    using namespace std;
    
    
    class trida {
    public:
        static int poleInt[3];
        static string poleString[3];
        static const char* poleConstChar[3];
    
        // Konstruktor
        trida() {
            poleInt[0] = 10;
            poleInt[1] = 20;
            poleInt[2] = 30;
    
            poleString[0] = "string: jedna";
            poleString[1] = "string: dva";
            poleString[2] = "string: tri";
    
            poleConstChar[0] = "const char*: jedna";
            poleConstChar[1] = "const char*: dva";
            poleConstChar[2] = "const char*: tri";
    
        }
    
        void zapisData(int zmenaInt, string zmenaString, const char* zmenaConstChar)
        {
            poleInt[1] = zmenaInt;
            poleString[1] = zmenaString;
            poleConstChar[1] = zmenaConstChar;
        }
    
        //zmeni jen 3. radek u polozky zmenaString
        void zapisDataStr(string zmenaString)
        {
            poleString[1] = zmenaString;
        }
    
        void vytiskniPole()
        {
            cout << poleInt[0] << endl;
            cout << poleInt[1] << endl;
            cout << poleInt[2] << endl;
    
            cout << endl;
    
            cout << poleString[0] << endl;
            cout << poleString[1] << endl;
            cout << poleString[2] << endl;
    
            cout << endl;
    
            cout << poleConstChar[0] << endl;
            cout << poleConstChar[1] << endl;
            cout << poleConstChar[2] << endl;
        }
    
    };
    
    
    //trida definuje jake hodnoty se maji zmenit - hodnoty se meni na primo, bez konverze
    class zmenitDataBezKonverze {
    public:
    
        trida mojeZmena;
    
        zmenitDataBezKonverze()
        {
            const char* zmenaConstChar = "const char*: zmana dat";
            int zmenaInt = -99;
            string zmenaString = "string: zmana dat";
    
            zmenData(zmenaInt, zmenaString, zmenaConstChar);
        }
    
        void zmenData(int zmenaInt, string zmenaString, const char* zmenaConstChar)
        {
            mojeZmena.zapisData(zmenaInt, zmenaString, zmenaConstChar);
        }
    
    };
    
    
    
    class zmenitDataSKonverzi {
    public:
    
        trida mojeZmenaKonverze;
    
        zmenitDataSKonverzi() {
    
            float hodnotaFloat = -0.99;
            string zmenaString = konvertujNaString(hodnotaFloat);
    
            zmenData(zmenaString);
    
        }
    
        string konvertujNaString(float hodnotaFloat)
        {
            string hodnotaFloatStr;
    
            std::ostringstream ss;
            ss << hodnotaFloat;
            hodnotaFloatStr = ss.str();
    
            return(hodnotaFloatStr);
        }
    
        //zmeni data jen u stringu
        void zmenData(string zmenaString)
        {
            mojeZmenaKonverze.zapisDataStr(zmenaString);
        }
    
    
    };
    
    
    
    // Inicializace statického člena třídy
    int trida::poleInt[3];
    string trida::poleString[3];
    const char* trida::poleConstChar[3];
    
    int main(void) {
    
        //zavola konstruktor hlavni tridy
        trida mojeTrida;
    
        //vytiskne pole
        mojeTrida.vytiskniPole();
    
        cout << endl << "----------------------" << endl;
        cout << "zmena dat z jine tridy bez konverze" << endl;
    
        //zapise data z jine tridy (data nekonvertuje)
        zmenitDataBezKonverze mojeDruhaTrida;
    
        //vytiskne pole
        mojeTrida.vytiskniPole();
    
        cout << endl << "----------------------" << endl;
        cout << "zmena dat z jine tridy s konverzi (pouze string)" << endl;
    
        //zkonvertuje float na string a zapise
        zmenitDataSKonverzi mojeTretiTrida;
    
        //vytiskne pole
        mojeTrida.vytiskniPole();
    
        cout << endl << endl << "Lze si vsimnout, ze se zmenil pouze string, int a char* const se nezmenil (neobsahuje zmeny z \"zmenitDataBezKonverze\"" << endl;
        cout << "To proto, ze data jsou pokazde znovu prepsana daty novymi, jelikoz data jsou definovany v konstruktoru \"trida\"" << endl;
        cout << "aby k opetovnemu prepsani dat nedochazelo, je potreba definovat data mimo konstruktor." << endl;
    
    
    
        return 0;
    }
